(function ( A ) {
	
	'use strict';

	var utils = A.utils, views = A.views;

	/**
	@module Anglebars
	*/

	/**
	Compiles a template string to an object, ready to be rendered
	
	@method compileTemplate
	@static
	@params template {string} The template to compile. Must be valid HTML
	@params preserveWhitespace=false {boolean} Whether to preserve whitespace between nodes.
	        Usually this whitespace is ignored by the browser, so can be omitted from the
	        compiled template
	@params replaceSrcAttributes=true {boolean} Whether to replace src attributes in the template
	        with data-anglebars-src to prevent the browser requesting non-existent resources
	        during compilation. Does not affect the output
	@return {object} The compiled template
	*/
	A.compileTemplate = function ( template, preserveWhitespace, replaceSrcAttributes ) {

		var nodes, stubs, compiled = [];

		// first, remove any comment mustaches
		template = utils.stripComments( template );

		// then, parse the template
		if ( replaceSrcAttributes === undefined ) { replaceSrcAttributes = true; }
		nodes = utils.getNodeArrayFromHtml( template, replaceSrcAttributes );
		
		// then, get an array of 'stubs' from the resulting DOM nodes
		stubs = utils.getStubsFromNodes( nodes );

		// finally, compile the stubs
		compiled = utils.compileStubs( stubs, 0, null, preserveWhitespace );

		return compiled;
	};


	A.render = function ( anglebars, el ) {

		var rendered;

		if ( !anglebars.compiled ) {
			throw new Error( 'No compiled template' );
		}

		rendered = new views.Fragment( anglebars.compiled, anglebars, el );

		return rendered;

	};



}( Anglebars ));
